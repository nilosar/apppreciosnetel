function renderCards(filter = "") { const resultsGrid = document.getElementById('resultsGrid'); resultsGrid.innerHTML = ""; const items = baseDeDatos.filter(item => String(item.modelo).toLowerCase().includes(filter.toLowerCase()) || String(item.sku).toLowerCase().includes(filter.toLowerCase()) ); if (items.length === 0) { document.getElementById('noResults').classList.remove('hidden'); return; } document.getElementById('noResults').classList.add('hidden'); items.forEach(item => { // Leemos los valores directamente del Excel sin cálculos extra const pNormal = Number(item.precio) || 0; const pPlan = Number(item.precio_plan) || 0; // Calculamos el ahorro solo para mostrarlo como info adicional const ahorro = pNormal - pPlan; resultsGrid.innerHTML += `
SKU: ${item.sku}

${item.modelo}

Precio Normal $${pNormal.toLocaleString('es-CL')}
Precio con Plan $${pPlan.toLocaleString('es-CL')}
${ahorro > 0 ? `
AHORRO -$${ahorro.toLocaleString('es-CL')}
` : ''}
`; }); }
